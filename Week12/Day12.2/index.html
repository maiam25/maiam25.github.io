Order by average (highest first) SELECT first_name, last_name, grade, average FROM students ORDER BY average DESC; 4. Only students with 80% or higher SELECT first_name, last_name, average FROM students WHERE average >= 80 ORDER BY average DESC; 5. Count students in each grade SELECT grade, COUNT(*) AS count FROM students GROUP BY grade ORDER BY grade; 6. Average mark by grade SELECT grade, AVG(average) AS avg_mark FROM students GROUP BY grade ORDER BY grade;
